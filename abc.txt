data _null_;
    /* Get the current date */
    current_date = today();
    
    /* Extract the year and month from the current date */
    year = year(current_date);
    month = month(current_date);
    
    /* Calculate the first day of the current month */
    first_day_of_month = mdy(month, 1, year);
    
    /* Find the day of the week for the first day of the month (1=Sunday, 2=Monday, ..., 7=Saturday) */
    day_of_week = weekday(first_day_of_month);
    
    /* Calculate the third Monday of the month */
    if day_of_week <= 2 then
        third_monday = first_day_of_month + (2 - day_of_week) + 14;
    else
        third_monday = first_day_of_month + (9 - day_of_week) + 14;
    
    /* Check if the current date is the third Monday of the month */
    if current_date = third_monday then do;
        /* Email settings */
        email_to = "recipient@example.com";
        email_subject = "Reminder: Third Monday of the Month";
        email_body = "Today is the third Monday of the month.";
        
        /* Send the email */
        filename mymail email to=email_to
                          subject=email_subject
                          type="text/plain";
        
        data _null_;
            file mymail;
            put email_body;
        run;
        
        filename mymail clear;
    end;
run;
